Creating DAX measures 16
Steps:
  1. Right click on the Aircraft crashes Table under the Data pane 
  2. From dropdown menu click on New measure 
  3. The formula bar appears along the top of the report canvas. 
  4.  Fill in all the DAX formulas provided below, one at a time, and when you finish Press Enter or select Commit (checkmark icon) in the formula bar.  
  5. After creating each measure choose the necessary format and decimal for each value (measure)
  6. Save


  1. Dax Formulas 
Total Crashes = COUNT ('Aircraft Crashes'[Date])
Crashes per Year =
                                  CALCULATE (
                                COUNTROWS ('Aircraft Crashes'),
                       ALLEXCEPT ('Calendar', 'Calendar'[Year (groups)])
    )
Total Fatalities = SUM ('Aircraft Crashes'[Total fatalities])
 Fatalities per Crash =
 DIVIDE (
    [Total Fatalities],
    [Total Crashes],
    0
)
Fatality Rate = SUM ('Aircraft Crashes'[Total fatalities]), COUNTROWS ('Aircraft Crashes') *100,0)
Total Survivors = FILTER ('Aircraft Crashes', 'Aircraft Crashes'[Survivors] = "Yes"))
Total People Involved = 
    SUM ('Aircraft Crashes'[Crew on board]) + SUM ('Aircraft Crashes'[Pax on board])
Survivor Rate = 
    DIVIDE (
        [Total Survivors],
        [Total People Involved],
        0
    )
YoY % Crashes =
VAR CrashesPreviousYear =
    CALCULATE (
        [Total Crashes],
        DATEADD('Calendar'[Date], -1, YEAR)
   )
VAR YoYCrashesRate = DIVIDE ([Total Crashes] - CrashesPreviousYear, CrashesPreviousYear, 0)
RETURN
    IF (
        ISBLANK(CrashesPreviousYear) || YoYCrashesRate = 0,
        BLANK (),
       YoYCrashesRate
    )
YoY %Fatalities = 
 
VAR PreviousYearFatalityRate = 
    CALCULATE (
        [Fatality Rate],
        SAMEPERIODLASTYEAR('Calendar'[Date])
    )
VAR YoYFatalityRate = 
    DIVIDE ([Fatality Rate] - PreviousYearFatalityRate, PreviousYearFatalityRate, 0)
RETURN
IF (
    ISBLANK(PreviousYearFatalityRate) || YoYFatalityRate = 0,
    BLANK (),
    YoYFatalityRate
)
YoY% Fatality Rate = 
VAR PreviousYearFatalityRate = 
    CALCULATE (
        [Fatality Rate],
        SAMEPERIODLASTYEAR('Calendar'[Date])
    )
VAR YoYFatalityRate = 
    DIVIDE ([Fatality Rate] - PreviousYearFatalityRate, PreviousYearFatalityRate, 0)
RETURN
IF (
    ISBLANK(PreviousYearFatalityRate) || YoYFatalityRate = 0,
    BLANK (),
    YoYFatalityRate
)
YOY%Survivor Rate = 
VAR CurrentYearSurvivability = [Survivor Rate]
VAR PreviousYearSurvivability = CALCULATE ([Survivor Rate], SAMEPERIODLASTYEAR('Calendar'[Date]))
VAR YoYSurvivorRate = DIVIDE (CurrentYearSurvivability - PreviousYearSurvivability, PreviousYearSurvivability, 0)
RETURN
    IF (
        ISBLANK(CurrentYearSurvivability) || ISBLANK(PreviousYearSurvivability) || YoYSurvivorRate = 0,
        BLANK (),
        YoYSurvivorRate
    )
Arrow YOY% Crashes = 

SWITCH (
    TRUE (), 
    [YoY % Crashes] < 0, UNICHAR (9662), 
    [YoY % Crashes] > 0, UNICHAR (9652), 
    BLANK () 
)
Arrow YOY Fatality Rate% = 
SWITCH (
    TRUE (), 
    [YoY% Fatality Rate] < 0, UNICHAR (9662), 
    [YoY% Fatality Rate] > 0, UNICHAR (9652), 
    BLANK () 
)
Arrow YOY% Fatalities = 
SWITCH (
    TRUE (), 
    ISBLANK ([YoY% Fatality Rate]), BLANK (), 
    [YoY% Fatality Rate] < 0, UNICHAR (9662),
    [YoY% Fatality Rate] > 0, UNICHAR (9652), 
    BLANK () 
)
Arrow% Survivor Rate = 
 SWITCH (
    TRUE (), 
    ISBLANK ([YOY%Survivor Rate]), BLANK (), 
    [YOY%Survivor Rate] < 0, UNICHAR (9662), 
    [YOY%Survivor Rate] > 0, UNICHAR (9652),
    BLANK () 
)
